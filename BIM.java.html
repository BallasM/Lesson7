import java.util.Scanner;

public class BIM {
    public static void main (String[] args) {
        Scanner userInput = new Scanner(System.in);
        System.out.println("Fulfill your weight in kg and then press the enter button: ");
        float myWeight = userInput.nextFloat();
        System.out.println("Fulfill you height in meters and then press the enter button: ");
        float myHeight = userInput.nextFloat();
        float bodyMassIndex = (float) (myWeight / Math.pow(myHeight,2));
        System.out.println("Your BMI is: "+ bodyMassIndex);
        if(bodyMassIndex < 18.5){
            System.out.println("You are underweight");
        } else if(bodyMassIndex > 18.5 && bodyMassIndex<= 24.9) {
            System.out.println("Your weight in normal");
        } else {
            System.out.println("You are overweight");
        }
    }
}
