import java.util.Scanner;

public class Triangle {
    public static void main(String[] args) {
        Scanner userInput = new Scanner(System.in);
        System.out.println("Kind of Triangle");
        System.out.println("Fulfill in cm the X side and then press Enter: ");
        int sideX = userInput.nextInt();
        System.out.println("Fulfill in cm the Y side and then press Enter: ");
        int sideY = userInput.nextInt();
        System.out.println("Fulfill in cm the Z side and then press Enter: ");
        int sideZ = userInput.nextInt();
        if(sideX == sideZ && sideY == sideZ && sideY == sideX){
            System.out.println("It is a Equilateral Triangle");

        }
        else if (sideX == sideY && sideY != sideZ && sideX != sideZ) {
            System.out.println("It is a Isosceles Triangle");
        } else {
            System.out.println("It is a Multifaceted Triangle");
            
        }


    }

}
